55 static AVLT_HEIGHT_t _node_height(avlt_node_t * node){
56  AVLT_HEIGHT_t height_left = 0;

60  }

124 static avlt_node_t * _node_balance(avlt_node_t * node){
125  avlt_node_t * newroot = NULL;
126 
127  /* Balance node's children, if they exist. */
128  if(node->left){
129  node->left = _node_balance(node->left);
130  }
131  if(node->right){
132  node->right = _node_balance(node->right);
133  }
134 
135  AVLT_HEIGHT_t bf = _node_balance_factor(node);
136 
137  if(bf >= 2){
138  /* Left Heavy */
139  if(_node_balance_factor(node->left) <= -1)
140  newroot = _node_rotate_leftright(node);
141  else
142  newroot = _node_rotate_leftleft(node);
143  }
144  else if(bf <= -2){
145  /* Right Heavy */
146  if(_node_balance_factor(node->right) >= 1)
147  newroot = _node_rotate_rightleft(node);
148  else
149  newroot = _node_rotate_rightright(node);
150  }
151  else {
152  /* This node is balanced -- no change. */
153  newroot = node;
154  }
155 
156  return(newroot);
157 }

167 void avlt_insert_node(avlt_t * tree, avlt_node_t * node){
168  avlt_node_t *next = NULL;
169  avlt_node_t *last = NULL;
170  if(tree->root == NULL){
171  /* Well, there must be a first node */
172  tree->root = node;
173  }
174  else{
175  /* Okay. We have a root already. Where do we put this? */
176  next = tree->root;
177  while(next != NULL){
178  last = next;
179  if(node->key < next->key){
180  next = next->left;
181  }
182  else if(node->key > next->key){
183  next = next->right;
184  }
185  else if(node->key == next->key) {
186  /* Have we already inserted this node? */
187  /* This shouldn't happen. */
188  }
189  }
190  if(node->key < last->key){
191  last->left = node;
192  }
193  if(node->key > last->key){
194  last->right = node;
195  }
196  }
197  _tree_balance(tree);
198 }
