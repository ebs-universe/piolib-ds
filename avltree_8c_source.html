55static AVLT_HEIGHT_t _node_height(avlt_node_t * node){
56 AVLT_HEIGHT_t height_left = 0;
57 AVLT_HEIGHT_t height_right = 0;
58 if( node->left ) {
59 height_left = _node_height( node->left );
60 }
61 if( node->right ) {
62 height_right = _node_height( node->right );
63 }
64 return height_right > height_left ? ++height_right : ++height_left;
65}

67static AVLT_HEIGHT_t _node_balance_factor(avlt_node_t * node){
68 AVLT_HEIGHT_t bf = 0;

70 bf += _node_height( node->left );

72 if(node->right){
73 bf -= _node_height( node->right );
74 }

124static avlt_node_t * _node_balance(avlt_node_t * node){
125 avlt_node_t * newroot = NULL;
126
127 /* Balance node's children, if they exist. */
128 if(node->left){
129 node->left = _node_balance(node->left);
130 }
131 if(node->right){
132 node->right = _node_balance(node->right);
133 }
134
135 AVLT_HEIGHT_t bf = _node_balance_factor(node);
136
137 if(bf >= 2){
138 /* Left Heavy */
139 if(_node_balance_factor(node->left) <= -1)
140 newroot = _node_rotate_leftright(node);
141 else
142 newroot = _node_rotate_leftleft(node);
143 }
144 else if(bf <= -2){
145 /* Right Heavy */
146 if(_node_balance_factor(node->right) >= 1)
147 newroot = _node_rotate_rightleft(node);
148 else
149 newroot = _node_rotate_rightright(node);
150 }
151 else {
152 /* This node is balanced -- no change. */
153 newroot = node;
154 }
155
156 return(newroot);
157}

167void avlt_insert_node(avlt_t * tree, avlt_node_t * node){
168 avlt_node_t *next = NULL;
169 avlt_node_t *last = NULL;
170 if(tree->root == NULL){
171 /* Well, there must be a first node */
172 tree->root = node;
173 }
174 else{
175 /* Okay. We have a root already. Where do we put this? */
176 next = tree->root;
177 while(next != NULL){
178 last = next;
179 if(node->key < next->key){
180 next = next->left;
181 }
182 else if(node->key > next->key){
183 next = next->right;
184 }
185 else if(node->key == next->key) {
186 /* Have we already inserted this node? */
187 /* This shouldn't happen. */
188 }
189 }
190 if(node->key < last->key){
191 last->left = node;
192 }
193 if(node->key > last->key){
194 last->right = node;
195 }
196 }
197 _tree_balance(tree);
198}

200avlt_node_t * avlt_find_node(avlt_t * tree, uint16_t key){
201 avlt_node_t * current = tree->root;
202 while(current && current->key!= key){
203 if(key > current->key)
204 current = current->right;
205 else
206 current = current->left;
207 }
