142static inline uint8_t bytebuf_cPushChunkLen(bytebuf * bytebufp, uint8_t * past_rollover){
143 uint8_t free = bytebufp->_size - bytebufp->_population;
144 uint8_t lend = bytebufp -> _bufp + bytebufp -> _size - bytebufp -> _inp + 1;
145 if (free < lend){
146 *past_rollover = 1;
147 return free;
148 }
149 else{
150 *past_rollover = 0;
151 return lend;
152 }
153 return 0;
154}
